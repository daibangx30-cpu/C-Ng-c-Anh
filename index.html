const GameSession = mongoose.model('GameSession', gameSessionSchema); // Question Schema const questionSchema = new mongoose.Schema({ zone: { type: Number, required: true, min: 1, max: 4 }, difficulty: { type: String, enum: ['easy', 'medium', 'hard'], default: 'medium' }, category: { type: String, default: 'mong_dac_thai_lien' }, question: { type: String, required: true }, options: [{ type: String, required: true }], correctAnswer: { type: String, required: true }, explanation: String, points: { type: Number, default: 10 } }); const Question = mongoose.model('Question', questionSchema); // ==================== SOCKET.IO SETUP ==================== const io = socketIO(server, { cors: { origin: process.env.CLIENT_URL || "*", methods: ["GET", "POST"], credentials: true }, transports: ['websocket', 'polling'], pingTimeout: 60000, pingInterval: 25000, cookie: false }); // ==================== REDIS SETUP ==================== let redisClient; let rateLimiter; (async () => { try { redisClient = createClient({ url: REDIS_URL }); await redisClient.connect(); console.log('✅ Redis connected'); } })(); // ==================== GAME STATE MANAGEMENT ==================== const activeRooms = new Map(); const playerSockets = new Map(); class GameEngine { constructor(roomId) { this.roomId = roomId; this.timers = new Map(); this.isProcessing = false; } async startGame(room) { try { const players = await Player.find({ roomId: room._id }); // Phân vai const wolfCount = room.settings.wolfCount || Math.max(1, Math.floor(players.length / 4)); 'wolf' : 'villager'; }); // Lưu vai trò await Promise.all(shuffledPlayers.map(p => p.save())); // Cập nhật trạng thái phòng room.gameStatus = 'playing'; room.startedAt = new Date(); room.gameData.currentPhase = 'day'; // Bắt đầu timer this.startGameTimer(room); return true; } catch (error) { console.error('Error starting game:', error); throw error; } } async submitAnswer(playerId, zone, questionIndex, answer) { try { const player = await Player.findById(playerId); const isCorrect = answer === question.correctAnswer; if (isCorrect) { // Cập nhật progress player.completedQuestions[`zone${zone}`] += 1; player.score += question.points; player.stats.correctAnswers += 1; await player.save(); // Gửi thông báo io.to(player.roomId.toString()).emit('answer-correct', { playerId: player._id, playerName: player.playerName, zone, points: question.points, streak: player.completedQuestions[`zone${zone}`] }); } } else { player.stats.wrongAnswers += 1; await player.save(); } return { correct: isCorrect, correctAnswer: question.correctAnswer }; } catch (error) { console.error('Error submitting answer:', error); throw error; } } async processVote(roomId, voterId, targetId) { try { const room = await GameRoom.findById(roomId); const voter = await Player.findById(voterId); const target = await Player.findById(targetId); } } async countVotes(room) { try { const players = await Player.find({ roomId: room._id, isAlive: true }); // Đếm votes const voteCount = {}; players.forEach(player => { if (player.voteTarget) { voteCount[player.voteTarget.toString()] = (voteCount[player.voteTarget.toString()] || 0) + 1; } }); // Tìm người bị vote nhiều nhất let maxVotes = 0; let accusedPlayerId = null; for (const [playerId, count] of Object.entries(voteCount)) { if (count > maxVotes) { maxVotes = count; accusedPlayerId = playerId; } } if (accusedPlayerId && maxVotes >= Math.ceil(players.length / 2)) { // Loại người chơi const accusedPlayer = await Player.findById(accusedPlayerId); if (accusedPlayer) { accusedPlayer.isAlive = false; await accusedPlayer.save(); // Lưu kết quả vote room.gameData.voteResults.push({ phase: `day-${room.gameData.dayCount}`, accusedPlayer: accusedPlayerId, voteCount: maxVotes, voters: players.filter(p => p.voteTarget?.toString() === accusedPlayerId).map(p => p._id) }); await room.save(); // Thông báo io.to(room._id.toString()).emit('player-eliminated', { playerId: accusedPlayer._id, playerName: accusedPlayer.playerName, reason: 'vote', role: accusedPlayer.role, voteCount: maxVotes }); // Kiểm tra điều kiện thắng await this.checkWinConditions(room); } } // Reset votes await Player.updateMany( { roomId: room._id }, { $set: { voteTarget: null } } ); return true; } catch (error) { console.error('Error counting votes:', error); throw error; } } async checkWinConditions(room) { try { const players = await Player.find({ roomId: room._id, isAlive: true }); const wolves = players.filter(p => p.role === 'wolf'); const villagers = players.filter(p => p.role !== 'wolf'); let winner = null; if (wolves.length === 0) { winner = 'villagers'; } else if (wolves.length >= villagers.length) { winner = 'wolves'; } if (winner) { await this.endGame(room, winner); } return winner; } catch (error) { console.error('Error checking win conditions:', error); throw error; } } async endGame(room, winner) { try { room.gameStatus = 'ended'; room.endedAt = new Date(); await room.save(); // Tính điểm và cập nhật thống kê const players = await Player.find({ roomId: room._id }); players.forEach(async (player) => { const didWin = (winner === 'wolves' && player.role === 'wolf') || (winner === 'villagers' && player.role !== 'wolf'); player.totalGames += 1; if (didWin) { player.gamesWon += 1; player.score += 100; // Bonus thắng } else { player.gamesLost += 1; } // Tính vote accuracy const alivePlayers = await Player.find({ roomId: room._id, isAlive: true }); const wolves = alivePlayers.filter(p => p.role === 'wolf'); if (player.voteTarget && wolves.length > 0) { const votedPlayer = await Player.findById(player.voteTarget); if (votedPlayer && votedPlayer.role === 'wolf') { player.stats.voteAccuracy = (player.stats.voteAccuracy + 1) / player.totalGames; } } await player.save(); }); // Tạo game session record const gameSession = new GameSession({ roomCode: room.roomCode, roomId: room._id, players: players.map(p => ({ playerId: p._id, playerName: p.playerName, role: p.role, finalScore: p.score, survived: p.isAlive, kills: p.stats.kills, correctAnswers: p.stats.correctAnswers })), winner, gameDuration: Math.floor((room.endedAt - room.startedAt) / 1000), startedAt: room.startedAt, endedAt: room.endedAt, settings: room.settings }); await gameSession.save(); // Gửi kết quả game io.to(room._id.toString()).emit('game-ended', { winner, players: players.map(p => ({ id: p._id, name: p.playerName, role: p.role, score: p.score, isAlive: p.isAlive, kills: p.stats.kills, correctAnswers: p.stats.correctAnswers })), duration: gameSession.gameDuration }); // Xóa room khỏi activeRooms activeRooms.delete(room._id.toString()); return true; } catch (error) { console.error('Error ending game:', error); throw error; } } startGameTimer(room) { const timer = setTimeout(async () => { try { const currentRoom = await GameRoom.findById(room._id); if (currentRoom && currentRoom.gameStatus === 'playing') { await this.endGame(currentRoom, 'draw'); // Hòa nếu hết giờ } } catch (error) { console.error('Error in game timer:', error); } }, room.settings.gameDuration * 1000); this.timers.set('game', timer); } } // ==================== SOCKET.IO EVENT HANDLERS ==================== io.use(async (socket, next) => { try { if (redisClient && rateLimiter) { await rateLimiter.consume(socket.handshake.address); } next(); } catch (error) { console.log('Rate limit exceeded for:', socket.handshake.address); next(new Error('Too many connection attempts')); } }); io.on('connection', (socket) => { console.log(`[${new Date().toISOString()}] Client connected: ${socket.id}`); // Heartbeat socket.on('heartbeat', () => { socket.emit('heartbeat-response', { timestamp: Date.now() }); }); // Tạo phòng socket.on('create-room', async ({ playerName, roomName, maxPlayers, settings }) => { try { // Tạo mã phòng ngẫu nhiên const roomCode = generateRoomCode(); } catch (error) { console.error('Error handling disconnect:', error); } }); }); // ==================== HELPER FUNCTIONS ==================== function generateRoomCode() { const chars = 'ABCDEFGHIJKLMNOPQRSTUVWXYZ0123456789'; let code = ''; for (let i = 0; i < 6; i++) { code += chars.charAt(Math.floor(Math.random() * chars.length)); } return code; } function filterMessage(message) { // Simple bad word filter const badWords = ['fuck', 'shit', 'asshole', 'bitch', 'damn'];
